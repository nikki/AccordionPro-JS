Note: when the linkable option is set, you can link to slides by using #yourDivId-slide-1, #yourDivId-slide-2, etc. in the link's href attribute.

  • Links:
  • #one-slide-1
  • #one-slide-2
  • #one-slide-3
  • #one-slide-4
  • #one-slide-5

Code:

<script>
    $('#one').accordionPro({
        theme : 'stitch',
        colour : {
            scheme : 'blue',
        },
        tab : {
            icon : 'chevron',
            font : 'Arial'
        },
        linkable : true
    });
</script>

Note: in this example, a callback function is created to show and hide the captions when the slides open and close. This is entirely optional.

Code:

<script>
    function slideOpen() {
        this.find('.ap-caption').delay(800).fadeIn();
    }

    function slideClose() {
        this.find('.ap-caption').fadeOut();
    }

    $('#two').accordionPro({
        autoPlay : true,
        onSlideOpen : slideOpen,
        onSlideClose : slideClose,
        theme : 'stitch',
        tab : {
            font : 'Helvetica Neue',
            icon : 'custom',
            customIcons : [
              '../documentation/img-demo/user-alt.png',
              '../documentation/img-demo/chat-alt.png',
              '../documentation/img-demo/letter-alt.png',
              '../documentation/img-demo/search-alt.png',
              '../documentation/img-demo/leaf-alt.png'
            ],
            customColours : [
              '#c25252',
              '#ca9859',
              '#96ba5f',
              '#59abb7',
              '#bb6098'
            ]
        },
        rounded : true
    });
</script>

Note: if you don't want images to be stretched to fit the slides as they are above, set the scaleImages option to false.

Code:

<script>
    $('#three').accordionPro({
        autoPlay : true,
        startClosed : true,
        theme : 'bordered',
        colour : {
            scheme : 'teal',
            style : 'gradient'
        },
        rounded : true
    });
</script>

Note: by popular request, you can now start the plugin in a closed position!

Code:

<script>
    $('#four').accordionPro({
        orientation : 'vertical',
        startClosed : true,
        theme : 'bordered',
        colour : {
            scheme : 'white',
            style : 'gradient'
        },
        verticalWidth : 600
    });
</script>
